#define DHTPIN 14
#define DHTTYPE DHT22
DHT dht(DHTPIN, DHTTYPE);

const int tempPin = 4;
OneWire oneWire(tempPin);
DallasTemperature tempSensor(&oneWire);

BH1750 lightMeter;
const int moisturePin = 32;

FirebaseData fbdo;
FirebaseAuth auth;
FirebaseConfig config;
bool signupOK = false;

void setup() {
  Serial.begin(115200);

  Wire.begin();
  lightMeter.begin();
  tempSensor.begin();
  dht.begin();

  Serial.print("Looking for Wi-Fi");
  WiFi.begin(WIFI_SSID, WIFI_PASSWORD);
  while (WiFi.status() != WL_CONNECTED) {
    Serial.print(".");
    delay(300);
  }
  Serial.println();
  Serial.print("Wi-Fi Connected! IP: ");
  Serial.println(WiFi.localIP());

  // Stop Wi-Fi from sleeping to prevent timeout errors
  WiFi.setSleep(false);

  Serial.println("Getting world time...");
  configTime(0, 0, "pool.ntp.org");
  delay(10000);

  config.api_key = API_KEY;
  config.database_url = DATABASE_URL;
  config.token_status_callback = tokenStatusCallback;

  Serial.println("Signing in...");

  // Using a test email and password to bypass the INVALID_EMAIL error
  if (Firebase.signUp(&config, &auth, "plant@test.com", "123456")) {
    signupOK = true;
  }
  Serial.println("Done!");

  fbdo.setBSSLBufferSize(4096, 1024);
  Firebase.begin(&config, &auth);
  Firebase.reconnectWiFi(true);
  Serial.println("Setup finished!");
}

void loop() {
  Serial.println("Reading light...");
  float lux = lightMeter.readLightLevel();

  Serial.println("Reading soil temp...");
  tempSensor.requestTemperatures();
  float soilTemp = tempSensor.getTempCByIndex(0);

  Serial.println("Reading moisture...");
  int rawMoisture = analogRead(moisturePin);
  int moisturePercent = map(rawMoisture, 3100, 2200, 0, 100);

  Serial.println("Reading air...");
  float airHum = dht.readHumidity();
  float airTemp = dht.readTemperature();

  Serial.println("Sending to cloud...");
  if (Firebase.ready() && signupOK) {
    Firebase.RTDB.setFloat(&fbdo, "sensor_data/light_lux", lux);
    Firebase.RTDB.setFloat(&fbdo, "sensor_data/soil_temp_c", soilTemp);
    Firebase.RTDB.setInt(&fbdo, "sensor_data/soil_moisture_percent", moisturePercent);
    Firebase.RTDB.setFloat(&fbdo, "sensor_data/air_temp_c", airTemp);
    Firebase.RTDB.setFloat(&fbdo, "sensor_data/air_humidity_percent", airHum);
    Serial.println("Sent to cloud!");
  } else {
    Serial.println("Waiting for cloud...");
  }

  delay(5000);
}
